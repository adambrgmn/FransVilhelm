# useEventListener

Will setup a listener on the window object and run handler every time that
action is triggered.

## Usage

```jsx
import React from 'react';
import { useEventListener } from '@adambrgmn/hooks';

const UseEventListenerExample = (): JSX.Element => {
  const [keys, setKeys] = React.useState<string[]>([]);
  useEventListener('keypress', ev => {
    setKeys([...keys, ev.key]);
  });

  return <output>{keys.join(' > ')}</output>;
};
```
